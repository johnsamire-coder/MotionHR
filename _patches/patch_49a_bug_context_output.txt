0006: from datetime import date, timedelta
0007: from django.db.models import Count, Sum, Avg, Q

0011: def get_date_range(period, custom_start=None, custom_end=None):
0012:     """
0013:     إرجاع نطاق التاريخ حسب الفترة
0014:     period: today, week, month, quarter, year, custom
0015:     """
0016:     today = date.today()
0017: 
0018:     if period == "today":
0019:         return today, today
0020: 
0021:     elif period == "week":
0022:         start = today - timedelta(days=today.weekday())
0023:         return start, today
0024: 
0025:     elif period == "month":
0026:         start = today.replace(day=1)
0027:         return start, today
0028: 
0029:     elif period == "last_month":
0030:         first_this = today.replace(day=1)
0031:         last_month_end = first_this - timedelta(days=1)
0032:         start = last_month_end.replace(day=1)
0033:         return start, last_month_end
0034: 
0035:     elif period == "quarter":
0036:         q_month = ((today.month - 1) // 3) * 3 + 1
0037:         start = today.replace(month=q_month, day=1)
0038:         return start, today
0039: 
0040:     elif period == "year":
0041:         start = today.replace(month=1, day=1)
0042:         return start, today
0043: 
0044:     elif period == "custom" and custom_start and custom_end:
0045:         return custom_start, custom_end
0046: 
0047:     # افتراضي: الشهر الحالي
0048:     return today.replace(day=1), today
0049:

0051: def get_attendance_summary(company, start_date, end_date, employee=None):
0052:     """ملخص الحضور في فترة معينة"""
0053:     from attendance.models import Attendance
0054: 
0055:     qs = Attendance.objects.filter(
0056:         company=company,
0057:         date__range=[start_date, end_date]
0058:     )
0059:     if employee:
0060:         qs = qs.filter(employee=employee)
0061: 
0062:     total     = qs.count()
0063:     present   = qs.filter(status="present").count()
0064:     absent    = qs.filter(status="absent").count()
0065:     late      = qs.filter(status="late").count()
0066:     on_leave  = qs.filter(status="on_leave").count()
0067:     holiday   = qs.filter(status__in=["holiday", "weekend"]).count()
0068: 
0069:     late_minutes_avg = qs.filter(
0070:         late_minutes__gt=0
0071:     ).aggregate(avg=Avg("late_minutes"))["avg"] or 0
0072: 
0073:     work_hours_total = qs.aggregate(
0074:         total=Sum("work_hours")
0075:     )["total"] or 0
0076: 
0077:     return {
0078:         "total":            total,
0079:         "present":          present,
0080:         "absent":           absent,
0081:         "late":             late,
0082:         "on_leave":         on_leave,
0083:         "holiday":          holiday,
0084:         "attendance_rate":  round(present / total * 100, 1) if total else 0,
0085:         "late_rate":        round(late / total * 100, 1) if total else 0,
0086:         "late_minutes_avg": round(late_minutes_avg, 0),
0087:         "work_hours_total": round(float(work_hours_total), 1),
0088:     }
